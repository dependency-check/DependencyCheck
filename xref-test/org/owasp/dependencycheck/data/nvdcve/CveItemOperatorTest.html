48      @Test
49      public void testTestCveCpeStartWithFilter() {
50  
51          ZonedDateTime published = ZonedDateTime.now();
52          ZonedDateTime lastModified = ZonedDateTime.now();
53          LocalDate cisaExploitAdd = null;
54          LocalDate cisaActionDue = null;
55          List<CveTag> cveTags = null;
56          List<LangString> descriptions = null;
57          List<Reference> references = null;
58          Metrics metrics = null;
59          List<Weakness> weaknesses = null;
60          List<Config> configurations = new ArrayList<>();
61          List<CpeMatch> matches = new ArrayList<>();
62          matches.add(null);
63          Node first = new Node(Node.Operator.OR, null, matches);
64          List<CpeMatch> cpeMatch = new ArrayList<>();
65          //cpe:2.3:o:microsoft:windows:-:*:*:*:*:*:*:*
66          CpeMatch match = new CpeMatch(true, "cpe:2.3:o:microsoft:windows:-:*:*:*:*:*:*:*", "matchCriteriaId", "versionStartExcluding",
67              "versionStartIncluding", "versionEndExcluding", "versionEndIncluding");
68          cpeMatch.add(match);
69          Node second = new Node(Node.Operator.OR, Boolean.FALSE, cpeMatch);
70          Node third = new Node(Node.Operator.OR, null, null);
71          List<Node> nodes = new ArrayList<>();
72          nodes.add(null);
73          nodes.add(first);
74          nodes.add(second);
75          nodes.add(third);
76          
77          Config c = new Config(Config.Operator.AND, null, nodes);
78          configurations.add(c);
79          List<VendorComment> vendorComments = null;
80          CveItem cveItem = new CveItem("id", "sourceIdentifier", "vulnStatus", published, lastModified,
81                  "evaluatorComment", "evaluatorSolution", "evaluatorImpact", cisaExploitAdd, cisaActionDue,
82                  "cisaRequiredAction", "cisaVulnerabilityName", cveTags, descriptions, references, metrics,
83                  weaknesses, configurations, vendorComments);
84  
85          DefCveItem cve = new DefCveItem(cveItem);
86          CveItemOperator instance = new CveItemOperator("cpe:2.3:o:");
87          boolean expResult = true;
88          boolean result = instance.testCveCpeStartWithFilter(cve);
89          assertEquals(expResult, result);
90  
91      }
