35      public void testException() throws Exception {
36  		boolean shouldFail = true;
37  		Dependency dependency10 = getDependency("1.0");
38  		Dependency dependency11 = getDependency("1.1");
39  		
40  		UnusedSuppressionRuleAnalyzer analyzer = getAnalyzer(shouldFail);
41  		Engine engine = getEngine(true, false, dependency10, dependency11);
42  		try {
43  			analyzer.analyzeDependency(dependency10, engine);
44  			throw new Exception("should have thrown an AnalysisException");
45  		} catch(AnalysisException ok){
46  			assertEquals(String.format(EXCEPTION_MSG, 1), ok.getMessage());
47  		}
48  		
49  		// no exception
50  		shouldFail = false;
51  		analyzer = getAnalyzer(shouldFail);
52  		engine = getEngine(true, false, dependency10, dependency11);
53  		analyzer.analyzeDependency(dependency10, engine);
54      	assertEquals(1, analyzer.getUnusedSuppressionRuleCount());

57      @Test
58      public void testCheckUnusedRules() throws Exception {
59  		// flag unset
60  		boolean shouldFail = false;
61  		Dependency dependency10 = getDependency("1.0");
62  		Dependency dependency11 = getDependency("1.1");
63  		
64  		// a run without any suppression rule ➫ no unused suppression
65  		checkUnusedRules(shouldFail, 0, false, false, dependency10);
66  
67  		// a run without no matching rule ➫ one unused suppression
68  		checkUnusedRules(shouldFail, 1, true, false, dependency10, dependency11);
69  
70  		// a run with the vulnerable package ➫ no unused suppression
71  		checkUnusedRules(shouldFail, 0, true, true, dependency10, dependency11);
72  
73  			
74  		// set flag
75  		shouldFail = true;
76  
77  		// a run without any suppression rule ➫ no unused suppression
78  		checkUnusedRules(shouldFail, 0, false, false, dependency10);
79  
80  		// a run without no matching rule ➫ one unused suppression
81  		checkUnusedRules(shouldFail, 1, true, false, dependency10, dependency11);
82  
83  		// a run with the vulnerable package ➫ no unused suppression
84  		checkUnusedRules(shouldFail, 0, true, true, dependency10, dependency11);
85      }

87  	private void checkUnusedRules(boolean shouldFail, int expectedCount, 
88  		boolean withSuppressionRules, boolean matching,
89  		Dependency ... dependencies) throws Exception {
90          UnusedSuppressionRuleAnalyzer analyzer = getAnalyzer(shouldFail);
91  		assertNotNull(analyzer);
92  		Engine engine = getEngine(withSuppressionRules, matching, dependencies);
93  		analyzer.checkUnusedRules(engine);
94  		assertEquals(expectedCount, analyzer.getUnusedSuppressionRuleCount());
95  	}

110 	private Engine getEngine(boolean hasSuppressionRules, boolean matching, Dependency ... dependencies) throws Exception {
111         Engine engine = new Engine(getSettings());
112 		List<Dependency> dependencyList = new ArrayList<>();
113 		if (dependencies!=null) {
114 			for(Dependency d : dependencies)
115 				dependencyList.add(d);
116 		}
117 		engine.setDependencies(dependencyList);
118 		if(!hasSuppressionRules) return engine;
119 		List<SuppressionRule> rules = new ArrayList<>();
120 		rules.add(getSuppressionRule(matching));
121 		engine.putObject(SUPPRESSION_OBJECT_KEY,rules);
122 		return engine;
123 	}
124 	
125 	private UnusedSuppressionRuleAnalyzer getAnalyzer(boolean flag) throws AnalysisException {
126         UnusedSuppressionRuleAnalyzer analyzer = new UnusedSuppressionRuleAnalyzer();
127 		assertNotNull(analyzer);
128 		
129 		Settings settings = getSettings();
130         settings.setBoolean(Settings.KEYS.FAIL_ON_UNUSED_SUPPRESSION_RULE, flag);
131         analyzer.initialize(settings);
132         assertEquals(flag, analyzer.failsForUnusedSuppressionRule());
133         assertEquals(0, analyzer.getUnusedSuppressionRuleCount());
134 		
135 		return analyzer;		
136 	}

138 	private SuppressionRule getSuppressionRule(boolean matching) {
139         SuppressionRule instance = new SuppressionRule();
140 		instance.addVulnerabilityName(getPropertyType("CVE-2023-5072", false, false));
141 		instance.setPackageUrl(getPropertyType("^pkg:maven/test." + PACKAGE_NAME + "." + matching, false, false));
142 		instance.addCpe(getPropertyType(PACKAGE_NAME, false, false));
143 		instance.setBase(false);
144 		instance.setMatched(matching);
145 		return instance;
146     }
